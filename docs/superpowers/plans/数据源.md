### 3.3 data_fetcher.py 核心实现

```python
import pandas as pd
import requests
from typing import Optional, List, Dict, Any
from datetime import datetime, timedelta
import time
import logging

logger = logging.getLogger(__name__)

# ============ 1. Tushare（A股）============
def fetch_a_share_stock_basic() -> pd.DataFrame:
    """获取所有 A 股基本信息"""
    import tushare as ts
    ts.set_token(config.TUSHARE_TOKEN)
    pro = ts.pro_api()
    df = pro.stock_basic(
        exchange='', list_status='L', 
        fields='ts_code,symbol,name,industry,market'
    )
    return df

def fetch_a_share_daily(ts_code: str, start_date: str, end_date: str) -> pd.DataFrame:
    """获取个股日K线数据"""
    import tushare as ts
    ts.set_token(config.TUSHARE_TOKEN)
    pro = ts.pro_api()
    df = pro.daily(ts_code=ts_code, start_date=start_date, end_date=end_date)
    return df

def fetch_a_share_financial(ts_code: str) -> pd.DataFrame:
    """获取个股财务指标（市盈率、净利润增长率等）"""
    import tushare as ts
    ts.set_token(config.TUSHARE_TOKEN)
    pro = ts.pro_api()
    df = pro.fina_indicator(ts_code=ts_code)
    return df

def fetch_a_share_low_pe_high_growth(
    pe_max: float = 20, 
    yoy_profit_min: float = 30
) -> pd.DataFrame:
    """筛选低市盈率、高增长的 A 股（供投研日报使用）"""
    import tushare as ts
    ts.set_token(config.TUSHARE_TOKEN)
    pro = ts.pro_api()
    
    # 获取全部 A 股列表
    stock_list = pro.stock_basic(exchange='', list_status='L')
    
    results = []
    for _, row in stock_list.iterrows():
        ts_code = row['ts_code']
        try:
            # 获取财务指标
            fin = pro.fina_indicator(ts_code=ts_code)
            if fin.empty:
                continue
            latest = fin.iloc[0]
            pe = latest.get('pe', None)
            yoy_profit = latest.get('yoy_profit', None)
            
            if pe and yoy_profit and pe > 0 and pe < pe_max and yoy_profit > yoy_profit_min:
                results.append({
                    'ts_code': ts_code,
                    'name': row['name'],
                    'industry': row['industry'],
                    'pe': pe,
                    'yoy_profit': yoy_profit
                })
        except Exception as e:
            logger.warning(f"获取 {ts_code} 数据失败: {e}")
            continue
        
        # 避免触发 API 频率限制
        time.sleep(0.5)
    
    return pd.DataFrame(results)


# ============ 2. Finnhub（美股）============
FINNHUB_BASE_URL = "https://finnhub.io/api/v1"

def finnhub_request(endpoint: str, params: Dict = None) -> Dict:
    """Finnhub API 通用请求"""
    if params is None:
        params = {}
    params['token'] = config.FINNHUB_API_KEY
    url = f"{FINNHUB_BASE_URL}/{endpoint}"
    resp = requests.get(url, params=params, timeout=30)
    resp.raise_for_status()
    return resp.json()

def fetch_us_stock_quote(symbol: str) -> Dict:
    """获取美股实时报价"""
    return finnhub_request("quote", {"symbol": symbol.upper()})

def fetch_us_company_profile(symbol: str) -> Dict:
    """获取公司基本信息"""
    return finnhub_request("stock/profile2", {"symbol": symbol.upper()})

def fetch_us_financials(symbol: str) -> Dict:
    """获取公司基本面财务数据（收入、利润、EPS 等）"""
    return finnhub_request("stock/financials", {
        "symbol": symbol.upper(),
        "statement": "ic"  # income statement
    })

def fetch_economic_calendar(from_date: str, to_date: str) -> List[Dict]:
    """获取宏观经济日历（利率决议、非农数据等）"""
    result = finnhub_request("calendar/economic", {
        "from": from_date,
        "to": to_date
    })
    return result.get("economicCalendar", [])


# ============ 3. CoinMarketCap（加密货币）============
CMC_BASE_URL = "https://pro-api.coinmarketcap.com/v1"

def cmc_request(endpoint: str, params: Dict = None) -> Dict:
    """CoinMarketCap API 通用请求"""
    if params is None:
        params = {}
    headers = {
        'X-CMC_PRO_API_KEY': config.COINMARKETCAP_API_KEY,
        'Accept': 'application/json'
    }
    url = f"{CMC_BASE_URL}/{endpoint}"
    resp = requests.get(url, headers=headers, params=params, timeout=30)
    resp.raise_for_status()
    return resp.json()

def fetch_crypto_listings(limit: int = 100) -> pd.DataFrame:
    """获取加密货币最新行情（价格、市值、成交量）"""
    data = cmc_request("cryptocurrency/listings/latest", {
        "limit": limit,
        "convert": "USD"
    })
    results = []
    for item in data.get('data', []):
        quote = item['quote']['USD']
        results.append({
            'name': item['name'],
            'symbol': item['symbol'],
            'price': quote['price'],
            'market_cap': quote['market_cap'],
            'volume_24h': quote['volume_24h'],
            'percent_change_24h': quote['percent_change_24h']
        })
    return pd.DataFrame(results)

def fetch_crypto_historical(symbol: str, days: int = 7) -> List[Dict]:
    """获取加密货币历史报价（需要 Hobbyist 或更高级别）"""
    # 注意：需要付费 tier 才能使用 historical endpoints
    # 免费 Key 仅支持 latest 接口
    logger.warning(f"历史报价需要付费 tier，当前仅返回模拟数据")
    return []


# ============ 4. Tavily（财经新闻搜索）============
TAVILY_BASE_URL = "https://api.tavily.com/search"

def search_finance_news(query: str, max_results: int = 10, days_back: int = 1) -> List[Dict]:
    """搜索财经新闻（AI 优化的搜索结果）"""
    payload = {
        "api_key": config.TAVILY_API_KEY,
        "query": query,
        "topic": "finance",          # 限定财经领域[reference:4]
        "search_depth": "basic",     # basic 足够，advanced 更贵
        "max_results": max_results,
        "time_range": "day" if days_back == 1 else "week",  # 可选: day, week, month[reference:5]
        "include_answer": False,     # 不需要 AI 回答，减少 token 消耗
        "include_raw_content": False # 不需要原始 HTML
    }
    resp = requests.post(TAVILY_BASE_URL, json=payload, timeout=60)
    resp.raise_for_status()
    data = resp.json()
    return data.get('results', [])

def fetch_hot_finance_topics() -> List[Dict]:
    """抓取当日财经热点（用于选题生成）"""
    # 组合多个查询，覆盖不同方向
    queries = [
        "A股 今日热点 重大新闻",
        "美股 市场动态",
        "宏观经济 政策 央行",
        "加密货币 市场 热点"
    ]
    all_results = []
    for q in queries:
        results = search_finance_news(q, max_results=5)
        all_results.extend(results)
    return all_results


# ============ 5. 统一数据采集入口 ============
def collect_market_summary() -> Dict[str, Any]:
    """采集全市场数据摘要（供选题生成使用）"""
    summary = {
        "a_share": {},
        "us_stock": {},
        "crypto": {},
        "news": []
    }
    
    # 1. A 股摘要：涨跌幅前5 + 低PE高增长筛选结果
    try:
        low_pe_stocks = fetch_a_share_low_pe_high_growth(pe_max=20, yoy_profit_min=30)
        summary["a_share"]["low_pe_high_growth_count"] = len(low_pe_stocks)
        summary["a_share"]["top_gainers"] = []  # 需要另行实现
    except Exception as e:
        logger.error(f"A股数据采集失败: {e}")
    
    # 2. 美股摘要：主要指数行情
    try:
        for idx, symbol in enumerate(["SPY", "QQQ", "AAPL", "NVDA"]):
            quote = fetch_us_stock_quote(symbol)
            summary["us_stock"][symbol] = {
                "price": quote.get("c"),  # current price
                "change_pct": quote.get("dp")  # change percent
            }
    except Exception as e:
        logger.error(f"美股数据采集失败: {e}")
    
    # 3. 加密货币摘要
    try:
        crypto_df = fetch_crypto_listings(limit=10)
        summary["crypto"]["top_10"] = crypto_df.to_dict('records')
    except Exception as e:
        logger.error(f"加密市场数据采集失败: {e}")
    
    # 4. 财经热点新闻
    try:
        summary["news"] = fetch_hot_finance_topics()
    except Exception as e:
        logger.error(f"新闻采集失败: {e}")
    
    return summary
```

## 四、降级与容错（Harness 层统一处理）

| 降级场景 | 触发条件 | 降级策略 | 成本影响 |
|----------|----------|----------|----------|
| Tushare 超限 | 积分不足/请求超频 | 切换 AkShare 免费爬虫 | 0 |
| Finnhub 超限 | 60次/分钟打满 | 降级使用 Yahoo Finance 兜底 | 0 |
| CoinMarketCap 付费墙 | 调用 historical 接口 | 仅用 latest 接口，历史数据走其他渠道 | 免费 |
| Tavily 配额用完 | 月度额度耗尽 | 改用 RSS 抓取或开源新闻 API | 需关注配额 |
| MinMaxi 降级 | 云端 API 不可用 | 回退到本地 Ollama（qwen2.5:7b） | 0 |

实现方式参考：

```python
def fetch_with_fallback(api_name: str, primary_func, fallback_func, *args, **kwargs):
    """统一降级装饰器"""
    try:
        return primary_func(*args, **kwargs)
    except Exception as e:
        logger.warning(f"{api_name} 主接口失败: {e}, 尝试降级")
        return fallback_func(*args, **kwargs)
```
