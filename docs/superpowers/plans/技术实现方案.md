## 3. 数据库设计（SQLite）

### 3.1 topics 表（选题库）
```sql
CREATE TABLE topics (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    title TEXT NOT NULL,                -- 选题标题
    angle TEXT,                         -- 核心角度
    hook_type TEXT,                     -- 冲突型/数据型/反常识型
    logic_type TEXT,                    -- 因果/数据/叙事/对比
    novelty_score REAL DEFAULT 0.5,    -- 新颖性（0-1）
    emotion_score REAL DEFAULT 0.5,     -- 情绪强度（0-1）
    source_article_id INTEGER,          -- 灵感来源爆文ID（预留）
    status TEXT DEFAULT 'pending',      -- pending / used / rejected
    created_date DATE DEFAULT (DATE('now')),
    used_count INTEGER DEFAULT 0,
    last_used_date DATE
);
```

### 3.2 posts 表（发布反馈）
```sql
CREATE TABLE posts (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    topic_id INTEGER,                   -- 关联 topics.id
    publish_date DATE,
    views INTEGER DEFAULT 0,
    likes INTEGER DEFAULT 0,
    comments INTEGER DEFAULT 0,
    url TEXT,                           -- 文章链接
    FOREIGN KEY (topic_id) REFERENCES topics(id)
);
```

### 3.3 traces 表（Harness 日志，用于 PDCA）
```sql
CREATE TABLE traces (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    trace_id TEXT UNIQUE,               -- UUID
    operation TEXT,                     -- 'generate_topic' / 'write_draft'
    model_name TEXT,
    input_summary TEXT,                 -- 输入前200字符
    output_summary TEXT,                -- 输出前200字符
    duration_ms INTEGER,
    success BOOLEAN,
    error_message TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
```

### 4.2 harness/local_model.py
```python
import requests
import json
import time
from config import LOCAL_OLLAMA_URL, LOCAL_MODEL_NAME

def generate(prompt: str, temperature=0.7, max_tokens=2000) -> str:
    """调用本地 Ollama 模型，返回生成的文本"""
    url = f"{LOCAL_OLLAMA_URL}/api/generate"
    payload = {
        "model": LOCAL_MODEL_NAME,
        "prompt": prompt,
        "stream": False,
        "options": {"temperature": temperature, "num_predict": max_tokens}
    }
    start = time.time()
    resp = requests.post(url, json=payload, timeout=120)
    elapsed = int((time.time() - start) * 1000)
    if resp.status_code == 200:
        return resp.json()["response"], elapsed
    else:
        raise Exception(f"Ollama error: {resp.text}")
```

### 4.3 harness/cloud_model.py（简化版，仅选题生成使用）
```python
import requests
import json
from config import CLOUD_API_KEY, CLOUD_MODEL_PROVIDER, CLOUD_MODEL_NAME

def generate(prompt: str, temperature=0.8) -> str:
    """调用云端模型（DeepSeek 示例）"""
    headers = {
        "Authorization": f"Bearer {CLOUD_API_KEY}",
        "Content-Type": "application/json"
    }
    payload = {
        "model": CLOUD_MODEL_NAME,
        "messages": [{"role": "user", "content": prompt}],
        "temperature": temperature
    }
    url = "https://api.deepseek.com/v1/chat/completions"
    resp = requests.post(url, headers=headers, json=payload, timeout=60)
    resp.raise_for_status()
    return resp.json()["choices"][0]["message"]["content"]
```

### 4.5 harness/core.py（Harness 核心封装）
```python
import uuid
import time
import json
from datetime import datetime
from db.crud import add_trace
from harness import local_model, cloud_model

def run_with_trace(operation, model_type, func, *args, **kwargs):
    """统一的 Harness 执行包装：记录 trace_id、耗时、成功/失败"""
    trace_id = str(uuid.uuid4())
    start = time.time()
    success = True
    error_msg = ""
    result = ""
    try:
        if model_type == "local":
            result, _ = func(*args, **kwargs)  # local_model.generate 返回 (text, elapsed)
        else:
            result = func(*args, **kwargs)
    except Exception as e:
        success = False
        error_msg = str(e)
        result = ""
    duration = int((time.time() - start) * 1000)
    # 异步记录到数据库（简单实现，不阻塞）
    add_trace(trace_id, operation, model_type, 
              input_summary=str(args)[:200], 
              output_summary=result[:200],
              duration_ms=duration, success=success, error_message=error_msg)
    if not success:
        raise Exception(f"Harness error: {error_msg}")
    return result, trace_id
```

### 4.6 pipelines/topic_generator.py
```python
import json
from harness.cloud_model import generate as cloud_generate
from harness.prompts import TOPIC_GENERATION_PROMPT
from harness.core import run_with_trace
from db.crud import add_topic

def fetch_hotspots():
    """获取当日热点（简化：从文件读取或调用 OpenClaw 输出）"""
    # 初期：手工创建文件 hot_today.txt，每行一个热点
    try:
        with open("hot_today.txt", "r", encoding="utf-8") as f:
            return f.read()
    except:
        return "暂无热点，请根据近期市场自行发挥。"

def fetch_historical_structures(limit=3):
    """从 topics 表读取历史高分选题作为结构参考"""
    # 简化：返回固定示例，或从数据库查询 novelty_score > 0.7 的选题
    return '[{"title": "比亚迪财报超预期，但风险在哪", "hook_type": "冲突型"}]'

def generate_topics(num=5):
    """生成选题并存入数据库"""
    hotspots = fetch_hotspots()
    structures = fetch_historical_structures()
    prompt = TOPIC_GENERATION_PROMPT.format(hotspots=hotspots, historical_structures=structures)
    response, trace_id = run_with_trace("generate_topics", "cloud", cloud_generate, prompt)
    topics = json.loads(response)  # 期望 JSON 数组
    for t in topics[:num]:
        add_topic(title=t["title"], angle=t["angle"], hook_type="", logic_type="", status="pending")
    return topics
```

### 4.7 pipelines/draft_writer.py
```python
from harness.local_model import generate as local_generate
from harness.prompts import DRAFT_WRITING_PROMPT
from harness.core import run_with_trace
from db.crud import get_pending_topic, update_topic_status
from config import DRAFTS_DIR
from datetime import datetime
import os

def write_draft_for_pending_topic():
    """从数据库中取一个 pending 选题，生成初稿，保存到 drafts/"""
    topic = get_pending_topic()
    if not topic:
        return None
    prompt = DRAFT_WRITING_PROMPT.format(title=topic[1], angle=topic[2])
    draft, trace_id = run_with_trace("write_draft", "local", local_generate, prompt)
    # 保存为 Markdown 文件
    date_str = datetime.now().strftime("%Y-%m-%d")
    filename = f"{date_str}_topic_{topic[0]}.md"
    filepath = os.path.join(DRAFTS_DIR, filename)
    with open(filepath, "w", encoding="utf-8") as f:
        f.write(f"# {topic[1]}\n\n")
        f.write(f"> 核心角度：{topic[2]}\n\n")
        f.write(draft)
        f.write(f"\n\n---\n*本文由 AI 生成初稿，人工审阅后发布，trace_id: {trace_id}*")
    # 更新 topic 状态为 used
    update_topic_status(topic[0], "used")
    return filepath
```

### 4.9 db/crud.py（数据库操作）
```python
import sqlite3
from config import DB_PATH

def get_connection():
    return sqlite3.connect(DB_PATH)

def init_db():
    """建表（在第一次运行时调用）"""
    with get_connection() as conn:
        conn.executescript("""
            CREATE TABLE IF NOT EXISTS topics (...);
            CREATE TABLE IF NOT EXISTS posts (...);
            CREATE TABLE IF NOT EXISTS traces (...);
        """)

def add_topic(title, angle, hook_type="", logic_type="", status="pending"):
    with get_connection() as conn:
        cur = conn.cursor()
        cur.execute(
            "INSERT INTO topics (title, angle, hook_type, logic_type, status) VALUES (?,?,?,?,?)",
            (title, angle, hook_type, logic_type, status)
        )
        return cur.lastrowid

def get_pending_topic():
    with get_connection() as conn:
        cur = conn.cursor()
        cur.execute("SELECT id, title, angle FROM topics WHERE status='pending' LIMIT 1")
        return cur.fetchone()

def update_topic_status(topic_id, status):
    with get_connection() as conn:
        conn.execute("UPDATE topics SET status=? WHERE id=?", (status, topic_id))

def add_trace(trace_id, operation, model_name, input_summary, output_summary, duration_ms, success, error_message=""):
    with get_connection() as conn:
        conn.execute(
            "INSERT INTO traces (trace_id, operation, model_name, input_summary, output_summary, duration_ms, success, error_message) VALUES (?,?,?,?,?,?,?,?)",
            (trace_id, operation, model_name, input_summary, output_summary, duration_ms, success, error_message)
        )
```

### 4.10 run.py（主入口，被 cron 调用）
```python
#!/usr/bin/env python3
import sys
import logging
from datetime import datetime
from pipelines.topic_generator import generate_topics
from pipelines.draft_writer import write_draft_for_pending_topic
from pipelines.daily_report import generate_research_report
from db.crud import init_db

# 配置简单日志
logging.basicConfig(level=logging.INFO, format='%(asctime)s - %(levelname)s - %(message)s')

def send_wechat_notification(text):
    """发送企业微信通知（可选）"""
    import requests
    from config import WECHAT_WEBHOOK_URL
    if not WECHAT_WEBHOOK_URL:
        return
    requests.post(WECHAT_WEBHOOK_URL, json={"msgtype": "text", "text": {"content": text}})

def main():
    logging.info("系统启动")
    init_db()
    
    # 1. 生成选题（每周跑几次，可以加环境变量控制）
    # 这里每天生成，但可以只生成新的 pending 选题
    topics = generate_topics(num=3)
    logging.info(f"生成了 {len(topics)} 个新选题")
    
    # 2. 基于一个 pending 选题生成初稿
    draft_path = write_draft_for_pending_topic()
    if draft_path:
        logging.info(f"初稿已生成：{draft_path}")
        send_wechat_notification(f"今日初稿已生成：{draft_path}")
    else:
        logging.warning("没有 pending 选题，请手工添加")
    
    # 3. 生成投研日报
    report_path = generate_research_report()
    logging.info(f"投研日报已生成：{report_path}")
    
    logging.info("系统执行完毕")

if __name__ == "__main__":
    main()
```

## 7. 总结

该方案是一个 **“刚好够用”** 的工程化实现：
- 保留了你的 Harness 哲学（所有调用经过封装、可观测、可降级）。
- 数据库仅用于选题和反馈，不滥用。
- 调度用 cron，避免 n8n 复杂度。
- 本地模型负责初稿，云端模型负责创意（选题）。
- 预留了 PDCA 接口（traces 表、posts 表）。

**你现在就可以把这个文档发给 OpenClaw / Trae，它应该能直接生成完整可运行的代码。**  
运行一周后，你只需手工往 `hot_today.txt` 写热点、往 `topics` 表插几个选题，系统就能每天自动产出初稿和投研日报。
